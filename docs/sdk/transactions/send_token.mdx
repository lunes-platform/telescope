    assetId="" # id of the token that should be sent, default token is Lunes

## Full Example

<Tabs>
  <TabItem value="Python3" label="Python3" default>

```python
from lunespy.tx.transfer import transfer_token_factory
from lunespy.wallet import wallet_factory

sender = wallet_factory()

tx = transfer_token_factory(
    # public key of sender
    sender=sender.public_key,
    # address that will receiver tokens
    recipient="37sEXsHBBUagA3jmzHPz1ZGA3vDtCLUNs5Z",
    # amount of tokens to send
    amount=249.99,
    # ID of the Token that should be sent, default Token is Lunes
    assetId="",
    # if you want to send a different fee amount, the default is 1000000
    fee=1000000

).sign(sender.private_key).broadcast(
    # url or ip:port of the node that will process the transaction,
    # defaul is https://lunesnode-testnet.lunes.io
    node = "https://my-full-node.com"
)
```
